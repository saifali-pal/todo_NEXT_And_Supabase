function Counter(){
    const [count, setCount] = useState(0)

    return (

        <div>
        <p>You Clicked: {count}</p>
        <button onclick={() => {setCount(count + 1)}}>Click here
        </button>
        <p>You clicked {count} times</p>
        </div>
    )
}
